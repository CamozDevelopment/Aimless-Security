Quick Start

Get protected in 3 lines of code

const Aimless = require('aimless-security');
const aimless = new Aimless({ rasp: { enabled: true } });
app.use(aimless.middleware());

✓ Your API is now protected against all major attack vectors!

Usage Examples

Basic Express Integration 

const express = require('express');
const Aimless = require('aimless-security');

const app = express();
app.use(express.json());

const aimless = new Aimless({
  rasp: {
    enabled: true,
    blockMode: true
  }
});

app.use(aimless.middleware());

app.get('/api/users', (req, res) => {
  res.json({ users: [] });
});

app.listen(3000);

CSRF Protection 

// Add CSRF middleware
app.use(aimless.csrf());

// CSRF token available in res.locals
app.get('/form', (req, res) => {
  res.send(`
    <form method="POST">
      <input type="hidden" 
        value="${res.locals.csrfToken}">
      <button>Submit</button>
    </form>
  `);
});

// Automatically validated
app.use(aimless.middleware());
app.post('/submit', (req, res) => {
  res.send('Success!');
});

XSS Sanitization 

app.post('/api/comments', (req, res) => {
  const { comment } = req.body;
  
  // Sanitize user input
  const safe = aimless.sanitize(comment);
  
  res.json({
    original: comment,
    sanitized: safe
  });
});

// Example:
// Input: <script>alert(1)</script>
// Output: &lt;script&gt;alert(1)&lt;/script&gt;

API Fuzzing 

const result = await aimless.fuzz({
  url: 'http://localhost:3000/api/users',
  method: 'POST',
  body: {
    username: 'test',
    password: 'test123'
  }
});

console.log(`Tested: ${result.testedPayloads}`);
console.log(`Found: ${result.vulnerabilities.length}`);

result.vulnerabilities.forEach(v => {
  console.log(`[${v.severity}] ${v.type}`);
});

Configuration

Full Configuration Options

const aimless = new Aimless({
  rasp: {
    enabled: true,                  // Enable/disable RASP
    blockMode: true,                // Block threats (false = monitor only)
    injectionProtection: true,      // SQL/NoSQL/Command injection
    xssProtection: true,            // XSS attack detection
    csrfProtection: true,           // CSRF protection
    anomalyDetection: true,         // Behavioral analysis
    trustedOrigins: [               // Trusted origins for CSRF
      'https://yourdomain.com'
    ],
    maxRequestSize: 10485760,      // Max request size (10MB)
    rateLimiting: {
      enabled: true,
      maxRequests: 100,            // Max requests per window
      windowMs: 60000              // Time window (1 minute)
    }
  },
  fuzzing: {
    enabled: true,
    maxPayloads: 100,              // Max payloads per parameter
    timeout: 5000,                 // Timeout per test (ms)
    authBypassTests: true,         // Test auth bypass
    rateLimitTests: true,          // Test rate limiting
    graphqlIntrospection: true,    // Test GraphQL
    customPayloads: []             // Custom payloads
  },
  logging: {
    enabled: true,
    level: 'info',                 // debug, info, warn, error
    logFile: './aimless.log'       // Optional log file
  }
});

API Methods

middleware()

Returns Express middleware for RASP protection

app.use(aimless.middleware());

csrf()

Returns CSRF protection middleware

app.use(aimless.csrf());

analyze(request)

Manually analyze a request for threats

const threats = aimless.analyze({
  method: 'POST',
  path: '/api/login',
  body: req.body,
  headers: req.headers,
  ip: req.ip
});

generateCSRFToken(sessionId)

Generate a CSRF token

const token = aimless.generateCSRFToken('session-123');

sanitize(output)

Sanitize output to prevent XSS

const safe = aimless.sanitize(userInput);

fuzz(target)

Fuzz test an API endpoint

const result = await aimless.fuzz({
  url: 'https://api.example.com/users',
  method: 'POST',
  body: { username: 'test' }
});
